   116→
   117→class DataGenerator:
   118→    """표준 라이브러리로 데이터셋 합성 생성 (numpy 불가)"""
   119→
   120→    @staticmethod
   121→    def _box_muller_gaussian():
   122→        """Box-Muller 변환으로 표준 정규분포 샘플 생성"""
   123→        u1 = random.random()
   124→        u2 = random.random()
   125→        z = math.sqrt(-2 * math.log(u1)) * math.cos(2 * math.pi * u2)
   126→        return z
   127→
   128→    def make_linear_regression(self, n: int, noise: float = 0.5) -> List[Sample]:
   129→        """선형 회귀 데이터: y = 2*x1 - 1*x2 + 0.5*x3 + bias + noise"""
   130→        data = []
   131→        for _ in range(n):
   132→            x1 = random.uniform(-5, 5)
   133→            x2 = random.uniform(-5, 5)
   134→            x3 = random.uniform(-5, 5)
   135→
   136→            y = 2 * x1 - 1 * x2 + 0.5 * x3 + 1.0
   137→            y += noise * self._box_muller_gaussian()
   138→
   139→            data.append(Sample(features=[x1, x2, x3], label=y))
   140→
   141→        return data
   142→
   143→    def make_binary_classification(self, n: int) -> List[Sample]:
   144→        """이진 분류 데이터: 2개 가우시안 클러스터"""
   145→        data = []
   146→
   147→        for _ in range(n):
   148→            if random.random() < 0.5:
   149→                # 클러스터 A: mean = (2, 2)
   150→                x1 = 2 + self._box_muller_gaussian()
   151→                x2 = 2 + self._box_muller_gaussian()
   152→                label = 0.0
   153→            else:
   154→                # 클러스터 B: mean = (-2, -2)
   155→                x1 = -2 + self._box_muller_gaussian()
   156→                x2 = -2 + self._box_muller_gaussian()
   157→                label = 1.0
   158→
   159→            data.append(Sample(features=[x1, x2], label=label))
   160→
   161→        return data
   162→
   163→    def make_xor_problem(self, n: int) -> List[Sample]:
   164→        """XOR 패턴 데이터 (선형 분리 불가 → MLP 필요성)"""
   165→        data = []
   166→
   167→        patterns = [
   168→            ([0.0, 0.0], 0.0),  # XOR: 0 ⊕ 0 = 0
   169→            ([1.0, 0.0], 1.0),  # XOR: 1 ⊕ 0 = 1
   170→            ([0.0, 1.0], 1.0),  # XOR: 0 ⊕ 1 = 1
   171→            ([1.0, 1.0], 0.0),  # XOR: 1 ⊕ 1 = 0
   172→        ]
   173→
   174→        for _ in range(n):
   175→            pattern, label = random.choice(patterns)
   176→            # 노이즈 추가 (0.1 표준편차)
   177→            x1 = pattern[0] + 0.1 * self._box_muller_gaussian()
   178→            x2 = pattern[1] + 0.1 * self._box_muller_gaussian()
   179→
   180→            data.append(Sample(features=[x1, x2], label=label))
   181→
   182→        return data
   183→
   184→    def train_test_split(self, data: List[Sample], ratio: float = 0.8
   185→                         ) -> Tuple[List[Sample], List[Sample]]:
   186→        """데이터 셔플 후 ratio 비율로 분리"""
   187→        shuffled = data.copy()
   188→        random.shuffle(shuffled)
   189→
   190→        split_idx = int(len(shuffled) * ratio)
   191→        return shuffled[:split_idx], shuffled[split_idx:]
   192→
   193→    def normalize(self, data: List[Sample]
   194→                  ) -> Tuple[List[Sample], List[float], List[float]]:
   195→        """z-score 정규화: (x - mean) / std"""
   196→        n_features = len(data[0].features)
   197→
   198→        # 평균 계산
   199→        means = [0.0] * n_features
   200→        for sample in data:
   201→            for i, x in enumerate(sample.features):
   202→                means[i] += x
   203→        means = [m / len(data) for m in means]
   204→
   205→        # 표준편차 계산
   206→        stds = [0.0] * n_features
   207→        for sample in data:
   208→            for i, x in enumerate(sample.features):
   209→                stds[i] += (x - means[i]) ** 2
   210→        stds = [math.sqrt(s / len(data)) for s in stds]
   211→
   212→        # 정규화
   213→        normalized_data = []
   214→        for sample in data:
   215→            normalized_features = []
   216→            for i, x in enumerate(sample.features):
   217→                if stds[i] > 1e-10:
   218→                    norm_x = (x - means[i]) / stds[i]
   219→                else:
   220→                    norm_x = 0.0
   221→                normalized_features.append(norm_x)
   222→
   223→            normalized_data.append(Sample(
   224→                features=normalized_features,
   225→                label=sample.label,
   226→                weight=sample.weight
   227→            ))
   228→
   229→        return normalized_data, means, stds
   230→

   232→# ═══════════════════════════════════════════════════════════════════════════
   233→# PART 2: LinearModel — 선형/로지스틱 회귀 + 경사하강법
   234→# ═══════════════════════════════════════════════════════════════════════════
   235→
   236→
   237→class LinearModel:
   238→    """선형/로지스틱 회귀 with 경사하강법"""
   239→
   240→    def __init__(self, n_features: int, model_type: str = "regression"):
   241→        """
   242→        Args:
   243→            n_features: 입력 특성 수
   244→            model_type: "regression" 또는 "logistic"
   245→        """
   246→        self.n_features = n_features
   247→        self.model_type = model_type
   248→
   249→        # 가중치 초기화: 작은 난수 (0.01 배율)
   250→        self.weights = [random.gauss(0, 0.01) for _ in range(n_features)]
   251→        self.bias = 0.0
   252→
   253→    def _sigmoid(self, z: float) -> float:
   254→        """시그모이드 함수: σ(z) = 1 / (1 + e^{-z})"""
   255→        if z > 500:
   256→            return 1.0
   257→        elif z < -500:
   258→            return 0.0
   259→        return 1.0 / (1.0 + math.exp(-z))
   260→
   261→    def _dot_product(self, x: List[float], w: List[float]) -> float:
   262→        """내적 계산"""
   263→        result = 0.0
   264→        for xi, wi in zip(x, w):
   265→            result += xi * wi
   266→        return result
   267→
   268→    def predict(self, x: List[float]) -> float:
   269→        """예측"""
   270→        z = self._dot_product(x, self.weights) + self.bias
   271→
   272→        if self.model_type == "regression":
   273→            return z
   274→        else:  # logistic
   275→            return self._sigmoid(z)
   276→
   277→    def compute_loss(self, samples: List[Sample]) -> float:
   278→        """손실함수 계산"""
   279→        n = len(samples)
   280→        total_loss = 0.0
   281→
   282→        for sample in samples:
   283→            y_hat = self.predict(sample.features)
   284→
   285→            if self.model_type == "regression":
   286→                # MSE: (y_hat - y)^2
   287→                loss = (y_hat - sample.label) ** 2
   288→            else:  # logistic
   289→                # BCE: -[y*log(y_hat) + (1-y)*log(1-y_hat)]
   290→                y = sample.label
   291→                eps = 1e-15
   292→                y_hat = max(eps, min(1 - eps, y_hat))
   293→                loss = -(y * math.log(y_hat) + (1 - y) * math.log(1 - y_hat))
   294→
   295→            total_loss += loss
   296→
   297→        return total_loss / n
   298→
   299→    def compute_gradients(self, samples: List[Sample]
   300→                         ) -> Tuple[List[float], float]:
   301→        """그래디언트 계산: dL/dw, dL/db"""
   302→        n = len(samples)
   303→        grad_w = [0.0] * self.n_features
   304→        grad_b = 0.0
   305→
   306→        for sample in samples:
   307→            y_hat = self.predict(sample.features)
   308→            y = sample.label
   309→
   310→            # 오류
   311→            error = y_hat - y
   312→
   313→            # 그래디언트 누적
   314→            for i in range(self.n_features):
   315→                grad_w[i] += error * sample.features[i]
   316→            grad_b += error
   317→
   318→        # 정규화
   319→        grad_w = [g / n for g in grad_w]
   320→        grad_b = grad_b / n
   321→
   322→        return grad_w, grad_b
   323→
   324→    def fit(self, samples: List[Sample], lr: float = 0.01,
   325→            epochs: int = 100) -> TrainingResult:
   326→        """경사하강법으로 학습"""
   327→        loss_history = []
   328→        accuracy_history = []
   329→
   330→        for epoch in range(epochs):
   331→            # 그래디언트 계산
   332→            grad_w, grad_b = self.compute_gradients(samples)
   333→
   334→            # 가중치 갱신
   335→            for i in range(self.n_features):
   336→                self.weights[i] -= lr * grad_w[i]
   337→            self.bias -= lr * grad_b
   338→
   339→            # 손실 기록
   340→            loss = self.compute_loss(samples)
   341→            loss_history.append(loss)
   342→
   343→            # 정확도 계산 (분류만)
   344→            if self.model_type == "logistic":
   345→                correct = sum(
   346→                    1 for s in samples
   347→                    if (self.predict(s.features) > 0.5) == (s.label > 0.5)
   348→                )
   349→                accuracy = correct / len(samples)
   350→                accuracy_history.append(accuracy)
   351→            else:
   352→                # 회귀: RMSE 사용
   353→                rmse = math.sqrt(loss)
   354→                accuracy_history.append(rmse)
   355→
   356→        return TrainingResult(
   357→            loss_history=loss_history,
   358→            accuracy_history=accuracy_history,
   359→            final_loss=loss_history[-1],
   360→            final_accuracy=accuracy_history[-1],
   361→            epochs_trained=epochs
   362→        )
   363→
   364→    def get_weights(self) -> ModelWeights:
   365→        """가중치를 ModelWeights 객체로 반환"""
   366→        # 선형 모델을 1층 신경망으로 표현
   367→        return ModelWeights(
   368→            weights=[[self.weights]],  # [layer=1][out_neuron=1][in=n_features]
   369→            biases=[[self.bias]],      # [layer=1][neuron=1]
   370→            layer_specs=[LayerSpec(
   371→                layer_type=LayerType.OUTPUT,
   372→                num_neurons=1,
   373→                activation=ActivationType.LINEAR
   374→            )]
   375→        )
   376→

   379→# PART 3: SGDOptimizer — SGD / 미니배치 / 모멘텀 최적화
   380→# ═══════════════════════════════════════════════════════════════════════════
   381→
   382→
   383→class SGDOptimizer:
   384→    """확률적 경사하강법 최적화"""
   385→
   386→    def __init__(self, lr: float = 0.01, optimizer_type: OptimizerType = OptimizerType.SGD,
   387→                 momentum: float = 0.9):
   388→        """
   389→        Args:
   390→            lr: 학습률
   391→            optimizer_type: SGD / MINI_BATCH / MOMENTUM
   392→            momentum: 모멘텀 계수 (0.9~0.99)
   393→        """
   394→        self.initial_lr = lr
   395→        self.lr = lr
   396→        self.optimizer_type = optimizer_type
   397→        self.momentum = momentum
   398→        self.velocity = {}  # 모멘텀 속도 저장
   399→
   400→    def get_mini_batches(self, data: List[Sample], batch_size: int
   401→                         ) -> List[List[Sample]]:
   402→        """데이터를 배치로 분할"""
   403→        shuffled = data.copy()
   404→        random.shuffle(shuffled)
   405→
   406→        batches = []
   407→        for i in range(0, len(shuffled), batch_size):
   408→            batches.append(shuffled[i:i + batch_size])
   409→
   410→        return batches
   411→
   412→    def lr_schedule(self, epoch: int, decay: float = 0.95) -> float:
   413→        """학습률 지수 감쇠"""
   414→        return self.initial_lr * (decay ** epoch)
   415→
   416→    def step(self, weights: List[float], gradients: List[float],
   417→             param_name: str = "default") -> List[float]:
   418→        """가중치 갱신"""
   419→        updated_weights = weights.copy()
   420→
   421→        if self.optimizer_type == OptimizerType.SGD:
   422→            # 순수 SGD: w ← w - lr·g
   423→            for i in range(len(updated_weights)):
   424→                updated_weights[i] -= self.lr * gradients[i]
   425→
   426→        elif self.optimizer_type == OptimizerType.MINI_BATCH:
   427→            # 미니배치도 동일 규칙
   428→            for i in range(len(updated_weights)):
   429→                updated_weights[i] -= self.lr * gradients[i]
   430→
   431→        elif self.optimizer_type == OptimizerType.MOMENTUM:
   432→            # 모멘텀: v ← β·v + lr·g, w ← w - v
   433→            if param_name not in self.velocity:
   434→                self.velocity[param_name] = [0.0] * len(gradients)
   435→
   436→            v = self.velocity[param_name]
   437→            for i in range(len(updated_weights)):
   438→                v[i] = self.momentum * v[i] + self.lr * gradients[i]
   439→                updated_weights[i] -= v[i]
   440→
   441→        return updated_weights
   442→
   443→    def run_mini_batch_training(self, model: LinearModel, data: List[Sample],
   444→                               batch_size: int, epochs: int) -> TrainingResult:
   445→        """미니배치 학습 루프"""
   446→        loss_history = []
   447→        accuracy_history = []
   448→
   449→        for epoch in range(epochs):
   450→            # 학습률 스케줄 적용
   451→            self.lr = self.lr_schedule(epoch)
   452→
   453→            # 미니배치 반복
   454→            batches = self.get_mini_batches(data, batch_size)
   455→            for batch in batches:
   456→                grad_w, grad_b = model.compute_gradients(batch)
   457→
   458→                # 가중치 갱신
   459→                model.weights = self.step(model.weights, grad_w, "weights")
   460→                model.bias -= self.lr * grad_b
   461→
   462→            # 손실 기록
   463→            loss = model.compute_loss(data)
   464→            loss_history.append(loss)
   465→
   466→            # 정확도 기록
   467→            if model.model_type == "logistic":
   468→                correct = sum(
   469→                    1 for s in data
   470→                    if (model.predict(s.features) > 0.5) == (s.label > 0.5)
   471→                )
   472→                accuracy = correct / len(data)
   473→                accuracy_history.append(accuracy)
   474→            else:
   475→                accuracy_history.append(math.sqrt(loss))
   476→
   477→        return TrainingResult(
   478→            loss_history=loss_history,
   479→            accuracy_history=accuracy_history,
   480→            final_loss=loss_history[-1],
   481→            final_accuracy=accuracy_history[-1],
   482→            epochs_trained=epochs
   483→        )

   487→# PART 4: NeuralLayer — 신경망 단일 레이어
   488→# ═══════════════════════════════════════════════════════════════════════════
   489→
   490→
   491→class NeuralLayer:
   492→    """신경망 레이어 (순전파 + 역전파 + 드롭아웃)"""
   493→
   494→    def __init__(self, spec: LayerSpec, n_inputs: int):
   495→        """
   496→        Args:
   497→            spec: LayerSpec (뉴런 수, 활성화 함수, 드롭아웃률)
   498→            n_inputs: 입력 차원
   499→        """
   500→        self.spec = spec
   501→        self.n_inputs = n_inputs
   502→        self.n_neurons = spec.num_neurons
   503→
   504→        # He 초기화: w ~ N(0, sqrt(2/fan_in))
   505→        std_init = math.sqrt(2.0 / n_inputs) if n_inputs > 0 else 0.01
   506→        self.weights = [
   507→            [random.gauss(0, std_init) for _ in range(n_inputs)]
   508→            for _ in range(self.n_neurons)
   509→        ]
   510→        self.biases = [0.0] * self.n_neurons
   511→
   512→        # 역전파용 그래디언트
   513→        self.dW = [[0.0] * n_inputs for _ in range(self.n_neurons)]
   514→        self.db = [0.0] * self.n_neurons
   515→
   516→        # 캐시 (역전파용)
   517→        self.last_x = None
   518→        self.last_z = None
   519→        self.last_a = None
   520→        self.dropout_mask = None
   521→
   522→    def _activate(self, z: float) -> float:
   523→        """활성화 함수"""
   524→        if self.spec.activation == ActivationType.RELU:
   525→            return max(0.0, z)
   526→        elif self.spec.activation == ActivationType.SIGMOID:
   527→            z = max(-500, min(500, z))
   528→            return 1.0 / (1.0 + math.exp(-z))
   529→        elif self.spec.activation == ActivationType.TANH:
   530→            return math.tanh(z)
   531→        else:  # LINEAR
   532→            return z
   533→
   534→    def _activate_derivative(self, z: float) -> float:
   535→        """활성화 함수 미분"""
   536→        if self.spec.activation == ActivationType.RELU:
   537→            return 1.0 if z > 0 else 0.0
   538→        elif self.spec.activation == ActivationType.SIGMOID:
   539→            a = self._activate(z)
   540→            return a * (1 - a)
   541→        elif self.spec.activation == ActivationType.TANH:
   542→            a = self._activate(z)
   543→            return 1 - a * a
   544→        else:  # LINEAR
   545→            return 1.0
   546→
   547→    def forward(self, x: List[float], training: bool = True) -> List[float]:
   548→        """순전파"""
   549→        self.last_x = x
   550→
   551→        # z = W·x + b
   552→        z = []
   553→        for i in range(self.n_neurons):
   554→            zi = self.biases[i]
   555→            for j in range(len(x)):
   556→                zi += self.weights[i][j] * x[j]
   557→            z.append(zi)
   558→
   559→        self.last_z = z
   560→
   561→        # 활성화
   562→        a = [self._activate(zi) for zi in z]
   563→
   564→        # 드롭아웃 (학습 시만)
   565→        if training and self.spec.dropout_rate > 0:
   566→            self.dropout_mask = [
   567→                random.random() > self.spec.dropout_rate
   568→                for _ in range(self.n_neurons)
   569→            ]
   570→            scale = 1.0 / (1.0 - self.spec.dropout_rate) if self.spec.dropout_rate < 1 else 1.0
   571→            a = [
   572→                a[i] * scale * (1 if self.dropout_mask[i] else 0)
   573→                for i in range(self.n_neurons)
   574→            ]
   575→
   576→        self.last_a = a
   577→        return a
   578→
   579→    def backward(self, delta: List[float], x: List[float]) -> List[float]:
   580→        """역전파 (delta: 이 레이어의 손실 신호)"""
   581→        # dW = delta · x^T
   582→        for i in range(self.n_neurons):
   583→            self.db[i] = delta[i]
   584→            for j in range(len(x)):
   585→                self.dW[i][j] = delta[i] * x[j]
   586→
   587→        # dx = W^T · delta (이전 레이어로 전달)
   588→        dx = [0.0] * len(x)
   589→        for j in range(len(x)):
   590→            for i in range(self.n_neurons):
   591→                dx[j] += self.weights[i][j] * delta[i]
   592→
   593→        return dx
   594→
   595→    def update_weights(self, lr: float) -> None:
   596→        """가중치 갱신"""
   597→        for i in range(self.n_neurons):
   598→            self.biases[i] -= lr * self.db[i]
   599→            for j in range(self.n_inputs):
   600→                self.weights[i][j] -= lr * self.dW[i][j]
   601→
   602→
   603→# ═══════════════════════════════════════════════════════════════════════════
   604→# PART 5: MLPNetwork — 다층 퍼셉트론
   605→# ═══════════════════════════════════════════════════════════════════════════
   606→
   607→
   608→class MLPNetwork:
   609→    """다층 퍼셉트론 (완전 연결 신경망)"""
   610→
   611→    def __init__(self, layer_specs: List[LayerSpec]):
   612→        """
   613→        Args:
   614→            layer_specs: 각 레이어 명세 (첫 번째 입력 차원 자동 결정)
   615→        """
   616→        self.layer_specs = layer_specs
   617→        self.layers = []
   618→
   619→        # 첫 입력 차원 설정 (임시, forward()에서 정해짐)
   620→        self.n_inputs = None
   621→
   622→    def _build_layers(self, n_inputs: int) -> None:
   623→        """첫 forward() 호출 시 레이어 구축"""
   624→        self.n_inputs = n_inputs
   625→        self.layers = []
   626→
   627→        for i, spec in enumerate(self.layer_specs):
   628→            if i == 0:
   629→                layer = NeuralLayer(spec, n_inputs)
   630→            else:
   631→                layer = NeuralLayer(spec, self.layer_specs[i-1].num_neurons)
   632→            self.layers.append(layer)
   633→
   634→    def forward(self, x: List[float], training: bool = True) -> List[float]:
   635→        """순전파"""
   636→        if not self.layers:
   637→            self._build_layers(len(x))
   638→
   639→        activations = x
   640→        for layer in self.layers:
   641→            activations = layer.forward(activations, training=training)
   642→
   643→        return activations
   644→
   645→    def backward(self, y_hat: List[float], y: float) -> None:
   646→        """역전파 (MSE 손실함수 기준)"""
   647→        # 출력층 델타: δ_L = (ŷ - y) · f'(z_L)
   648→        output_layer = self.layers[-1]
   649→        delta = []
   650→        for i in range(output_layer.n_neurons):
   651→            dz = (y_hat[i] - y) * output_layer._activate_derivative(output_layer.last_z[i])
   652→            delta.append(dz)
   653→
   654→        # 역방향 역전파
   655→        for l in range(len(self.layers) - 1, -1, -1):
   656→            layer = self.layers[l]
   657→
   658→            if l > 0:
   659→                prev_layer = self.layers[l - 1]
   660→                input_to_layer = prev_layer.last_a
   661→            else:
   662→                input_to_layer = layer.last_x
   663→
   664→            # 그래디언트 계산
   665→            layer.backward(delta, input_to_layer)
   666→
   667→            # 이전 레이어로 역전파할 델타 계산
   668→            if l > 0:
   669→                prev_layer = self.layers[l - 1]
   670→                delta = layer.backward(delta, input_to_layer)
   671→
   672→                # 활성화 함수 미분 적용
   673→                new_delta = []
   674→                for i in range(prev_layer.n_neurons):
   675→                    dz = delta[i] * prev_layer._activate_derivative(prev_layer.last_z[i])
   676→                    new_delta.append(dz)
   677→                delta = new_delta
   678→
   679→    def update_weights(self, lr: float) -> None:
   680→        """가중치 갱신"""
   681→        for layer in self.layers:
   682→            layer.update_weights(lr)
   683→
   684→    def compute_loss(self, samples: List[Sample]) -> float:
   685→        """손실함수 계산 (MSE)"""
   686→        total_loss = 0.0
   687→        for sample in samples:
   688→            y_hat = self.forward(sample.features, training=False)
   689→            loss = (y_hat[0] - sample.label) ** 2
   690→            total_loss += loss
   691→        return total_loss / len(samples)
   692→
   693→    def fit(self, samples: List[Sample], lr: float = 0.01,
   694→            epochs: int = 100, batch_size: int = 32) -> TrainingResult:
   695→        """미니배치 학습"""
   696→        loss_history = []
   697→        accuracy_history = []
   698→
   699→        for epoch in range(epochs):
   700→            # 미니배치 생성
   701→            shuffled = samples.copy()
   702→            random.shuffle(shuffled)
   703→
   704→            for i in range(0, len(shuffled), batch_size):
   705→                batch = shuffled[i:i + batch_size]
   706→
   707→                # 배치 학습
   708→                for sample in batch:
   709→                    y_hat = self.forward(sample.features, training=True)
   710→                    self.backward(y_hat, sample.label)
   711→                    self.update_weights(lr)
   712→
   713→            # 손실 기록
   714→            loss = self.compute_loss(samples)
   715→            loss_history.append(loss)
   716→
   717→            # 정확도 (분류: 0.5 기준)
   718→            correct = 0
   719→            for sample in samples:
   720→                y_hat = self.forward(sample.features, training=False)
   721→                pred = 1.0 if y_hat[0] > 0.5 else 0.0
   722→                if pred == sample.label:
   723→                    correct += 1
   724→
   725→            accuracy = correct / len(samples)
   726→            accuracy_history.append(accuracy)
   727→
   728→        return TrainingResult(
   729→            loss_history=loss_history,
   730→            accuracy_history=accuracy_history,
   731→            final_loss=loss_history[-1],
   732→            final_accuracy=accuracy_history[-1],
   733→            epochs_trained=epochs
   734→        )
   735→
   736→    def predict_batch(self, samples: List[Sample]) -> List[float]:
   737→        """배치 예측"""
   738→        predictions = []
   739→        for sample in samples:
   740→            y_hat = self.forward(sample.features, training=False)
   741→            predictions.append(y_hat[0])
   742→        return predictions
   743→
   744→
   745→# ═══════════════════════════════════════════════════════════════════════════
   746→# PART 6: GradientCompressor — 그래디언트 압축
   747→# ═══════════════════════════════════════════════════════════════════════════
   748→
   749→
   750→class GradientCompressor:
   751→    """그래디언트 압축 (Top-k 희소화 + 양자화)"""
   752→
   753→    def __init__(self, compression_ratio: float = 0.1):
   754→        """
   755→        Args:
   756→            compression_ratio: 전송할 그래디언트 비율 (0~1)
   757→        """
   758→        self.compression_ratio = compression_ratio
   759→
   760→    def top_k_sparsification(self, gradients: List[float]) -> Dict[int, float]:
   761→        """상위 k개 그래디언트만 선택 (희소 표현)"""
   762→        k = max(1, int(len(gradients) * self.compression_ratio))
   763→
   764→        # (인덱스, 절대값) 튜플로 정렬
   765→        indexed = [(i, abs(g)) for i, g in enumerate(gradients)]
   766→        indexed.sort(key=lambda x: x[1], reverse=True)
   767→
   768→        # 상위 k개만 반환
   769→        sparse_dict = {}
   770→        for i in range(min(k, len(indexed))):
   771→            idx, _ = indexed[i]
   772→            sparse_dict[idx] = gradients[idx]
   773→
   774→        return sparse_dict
   775→
   776→    def quantize(self, gradients: List[float], bits: int = 8) -> Tuple[List[int], float, float]:
   777→        """양자화: 실수 → 정수"""
   778→        if not gradients:
   779→            return [], 0.0, 1.0
   780→
   781→        min_val = min(gradients)
   782→        max_val = max(gradients)
   783→
   784→        if abs(max_val - min_val) < 1e-10:
   785→            return [0] * len(gradients), min_val, 1.0
   786→
   787→        # 양자화 스케일
   788→        scale = (max_val - min_val) / (2 ** bits - 1)
   789→
   790→        quantized = []
   791→        for g in gradients:
   792→            q = int(round((g - min_val) / scale))
   793→            q = max(0, min(2 ** bits - 1, q))
   794→            quantized.append(q)
   795→
   796→        return quantized, min_val, scale
   797→
   798→    def dequantize(self, quantized: List[int], min_val: float,
   799→                   scale: float) -> List[float]:
   800→        """역양자화"""
   801→        return [q * scale + min_val for q in quantized]
   802→
   803→    def compress(self, gradients: List[float]) -> Tuple[Dict, float]:
   804→        """그래디언트 압축"""
   805→        # Step 1: Top-k 희소화
   806→        sparse_dict = self.top_k_sparsification(gradients)
   807→
   808→        # Step 2: 희소 값들만 양자화
   809→        sparse_values = list(sparse_dict.values())
   810→        if sparse_values:
   811→            quantized, min_val, scale = self.quantize(sparse_values, bits=8)
   812→
   813→            # 압축률 계산
   814→            original_size = len(gradients)
   815→            compressed_size = len(sparse_dict)
   816→            compression_ratio = compressed_size / original_size if original_size > 0 else 0
   817→
   818→            return {
   819→                "sparse_dict": sparse_dict,
   820→                "quantized": quantized,
   821→                "min_val": min_val,
   822→                "scale": scale,
   823→                "original_size": original_size
   824→            }, compression_ratio
   825→
   826→        return {}, 1.0
   827→
   828→    def decompress(self, compressed: Dict, original_size: int) -> List[float]:
   829→        """압축 해제"""
   830→        if not compressed or "sparse_dict" not in compressed:
   831→            return [0.0] * original_size
   832→
   833→        sparse_dict = compressed["sparse_dict"]
   834→        quantized = compressed["quantized"]
   835→        min_val = compressed["min_val"]
   836→        scale = compressed["scale"]
   837→
   838→        # 역양자화
   839→        dequantized = self.dequantize(quantized, min_val, scale)
   840→
   841→        # 희소 표현 복원
   842→        result = [0.0] * original_size
   843→        for i, (idx, _) in enumerate(sparse_dict.items()):
   844→            if i < len(dequantized):
   845→                result[idx] = dequantized[i]
   846→
   847→        return result
   848→

   850→# ═══════════════════════════════════════════════════════════════════════════
   851→# PART 7: FederatedClient — 연합 학습 클라이언트
   852→# ═══════════════════════════════════════════════════════════════════════════
   853→
   854→
   855→class FederatedClient:
   856→    """Federated Learning 클라이언트"""
   857→
   858→    def __init__(self, client_id: str, local_data: List[Sample],
   859→                 layer_specs: List[LayerSpec]):
   860→        """
   861→        Args:
   862→            client_id: 클라이언트 식별자
   863→            local_data: 로컬 학습 데이터
   864→            layer_specs: 신경망 레이어 명세
   865→        """
   866→        self.client_id = client_id
   867→        self.local_data = local_data
   868→        self.layer_specs = layer_specs
   869→
   870→        # 로컬 신경망
   871→        self.local_model = MLPNetwork(layer_specs)
   872→
   873→        # 압축기
   874→        self.compressor = GradientCompressor(compression_ratio=0.1)
   875→
   876→        # 통계
   877→        self.last_loss = None
   878→
   879→    def local_train(self, global_weights: Optional[ModelWeights],
   880→                    lr: float = 0.01, local_epochs: int = 5) -> TrainingResult:
   881→        """로컬 데이터로 학습"""
   882→        # 글로벌 가중치 로드 (있으면)
   883→        if global_weights and global_weights.weights:
   884→            # 간단한 구현: 가중치 직접 로드 (복잡한 직렬화 생략)
   885→            pass
   886→
   887→        # 로컬 학습
   888→        result = self.local_model.fit(
   889→            self.local_data,
   890→            lr=lr,
   891→            epochs=local_epochs,
   892→            batch_size=min(16, len(self.local_data))
   893→        )
   894→
   895→        self.last_loss = result.final_loss
   896→        return result
   897→
   898→    def get_model_delta(self) -> List[float]:
   899→        """모델 업데이트 (델타 = 현재 - 글로벌)"""
   900→        # 간단한 구현: 현재 가중치 반환
   901→        delta = []
   902→        for layer in self.local_model.layers:
   903→            for weights in layer.weights:
   904→                delta.extend(weights)
   905→            delta.extend(layer.biases)
   906→        return delta
   907→
   908→    def compress_update(self, delta: List[float]) -> Tuple[Dict, float]:
   909→        """업데이트 압축"""
   910→        return self.compressor.compress(delta)
   911→
   912→    def report_stats(self) -> Dict[str, Any]:
   913→        """통계 보고"""
   914→        return {
   915→            "client_id": self.client_id,
   916→            "num_samples": len(self.local_data),
   917→            "last_loss": self.last_loss
   918→        }
   919→

   921→# ═══════════════════════════════════════════════════════════════════════════
   922→# PART 8: FederatedServer — 연합 학습 서버 (FedAvg)
   923→# ═══════════════════════════════════════════════════════════════════════════
   924→
   925→
   926→class FederatedServer:
   927→    """Federated Learning 서버 (FedAvg 알고리즘)"""
   928→
   929→    def __init__(self, layer_specs: List[LayerSpec], n_clients: int):
   930→        """
   931→        Args:
   932→            layer_specs: 신경망 명세
   933→            n_clients: 클라이언트 수
   934→        """
   935→        self.layer_specs = layer_specs
   936→        self.n_clients = n_clients
   937→
   938→        # 글로벌 신경망
   939→        self.global_model = MLPNetwork(layer_specs)
   940→
   941→        # FL 라운드 히스토리
   942→        self.round_history = []
   943→
   944→    def broadcast_weights(self) -> List[float]:
   945→        """글로벌 가중치 배포"""
   946→        weights = []
   947→        for layer in self.global_model.layers:
   948→            for w_row in layer.weights:
   949→                weights.extend(w_row)
   950→            weights.extend(layer.biases)
   951→        return weights
   952→
   953→    def aggregate_fed_avg(self, client_weights: List[Tuple[List[float], int]]
   954→                         ) -> List[float]:
   955→        """FedAvg 집계"""
   956→        # FedAvg: w_global = Σ(n_k / n_total) * w_k
   957→        total_samples = sum(n for _, n in client_weights)
   958→
   959→        if not client_weights or total_samples == 0:
   960→            return self.broadcast_weights()
   961→
   962→        aggregated = None
   963→        for weights, n in client_weights:
   964→            weight_ratio = n / total_samples
   965→
   966→            if aggregated is None:
   967→                aggregated = [w * weight_ratio for w in weights]
   968→            else:
   969→                for i in range(len(aggregated)):
   970→                    aggregated[i] += weights[i] * weight_ratio
   971→
   972→        return aggregated if aggregated is not None else self.broadcast_weights()
   973→
   974→    def evaluate_global_model(self, test_data: List[Sample]
   975→                             ) -> Tuple[float, float]:
   976→        """글로벌 모델 평가"""
   977→        loss = self.global_model.compute_loss(test_data)
   978→
   979→        # 정확도
   980→        correct = 0
   981→        for sample in test_data:
   982→            y_hat = self.global_model.forward(sample.features, training=False)
   983→            pred = 1.0 if y_hat[0] > 0.5 else 0.0
   984→            if pred == sample.label:
   985→                correct += 1
   986→
   987→        accuracy = correct / len(test_data) if test_data else 0.0
   988→        return loss, accuracy
   989→
   990→    def run_round(self, clients: List[FederatedClient], lr: float = 0.01,
   991→                  local_epochs: int = 5, select_fraction: float = 1.0
   992→                  ) -> FederatedRound:
   993→        """FL 한 라운드 실행"""
   994→        round_num = len(self.round_history) + 1
   995→
   996→        # 클라이언트 선택
   997→        n_selected = max(1, int(len(clients) * select_fraction))
   998→        selected_clients = random.sample(clients, n_selected)
   999→
  1000→        # 글로벌 가중치 배포
  1001→        global_weights = self.broadcast_weights()
  1002→
  1003→        # 로컬 학습
  1004→        client_updates = []
  1005→        client_losses = []
  1006→
  1007→        for client in selected_clients:
  1008→            # 간단한 구현: 직접 로컬 학습
