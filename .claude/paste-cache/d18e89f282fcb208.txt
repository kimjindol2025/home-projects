1. v14.0 설계 목표: "텍스트에서 의미를 추출하다"
 * Tokenization: let x = 10;이라는 단순한 문자열을 [Let, Ident("x"), Assign, Int(10), Semicolon]이라는 의미 있는 묶음으로 분리합니다.
 * 공백 및 주석 제거: 컴파일에 불필요한 공백, 탭, 주석을 필터링하여 엔진의 부하를 줄입니다.
 * 에러 탐지: 정의되지 않은 기호나 닫히지 않은 문자열을 초기에 잡아냅니다.
2. 설계자의 관점: "유한 상태 기계 (Finite State Machine)"
Lexer는 소스코드를 한 글자씩 읽으며 현재 상태에 따라 다음 토큰을 결정하는 거대한 기계와 같습니다. "숫자를 읽는 중인가?", "식별자를 읽는 중인가?"를 끊임없이 판단해야 합니다.

// [제13장: 컴파일러 구현 v14.0 - Lexer 설계]

#[derive(Debug, PartialEq)]
enum Token {
    Let,
    Ident(String),
    Assign,
    Int(i32),
    Semicolon,
    Plus,
    EOF, // End Of File
}

struct Lexer {
    input: Vec<char>,
    position: usize,      // 현재 읽고 있는 위치
    read_position: usize, // 다음 글자 위치
    ch: char,             // 현재 조사 중인 문자
}

impl Lexer {
    fn new(input: String) -> Self {
        let mut l = Lexer {
            input: input.chars().collect(),
            position: 0,
            read_position: 0,
            ch: '\0',
        };
        l.read_char();
        l
    }

    fn read_char(&mut self) {
        if self.read_position >= self.input.len() {
            self.ch = '\0';
        } else {
            self.ch = self.input[self.read_position];
        }
        self.position = self.read_position;
        self.read_position += 1;
    }

    fn next_token(&mut self) -> Token {
        self.skip_whitespace();

        let token = match self.ch {
            '=' => Token::Assign,
            '+' => Token::Plus,
            ';' => Token::Semicolon,
            '\0' => Token::EOF,
            _ => {
                if self.ch.is_alphabetic() {
                    let ident = self.read_identifier();
                    return match ident.as_str() {
                        "let" => Token::Let,
                        _ => Token::Ident(ident),
                    };
                } else if self.ch.is_numeric() {
                    return Token::Int(self.read_number());
                } else {
                    Token::EOF
                }
            }
        };

        self.read_char();
        token
    }

    fn skip_whitespace(&mut self) {
        while self.ch.is_whitespace() {
            self.read_char();
        }
    }

    fn read_identifier(&mut self) -> String {
        let pos = self.position;
        while self.ch.is_alphabetic() {
            self.read_char();
        }
        self.input[pos..self.position].iter().collect()
    }

    fn read_number(&mut self) -> i32 {
        let pos = self.position;
        while self.ch.is_numeric() {
            self.read_char();
        }
        let s: String = self.input[pos..self.position].iter().collect();
        s.parse().unwrap_or(0)
    }
}

fn main() {
    println!("--- v14.0 Compiler Frontend: The Lexer ---");

    let input = "let x = 5 + 10;".to_string();
    let mut lexer = Lexer::new(input);

    loop {
        let tok = lexer.next_token();
        println!("  [TOKEN] {:?}", tok);
        if tok == Token::EOF { break; }
    }

    println!("\n기록: 소스코드를 토큰화하여 컴파일러 엔진의 입력을 정형화함.");
    println!("저장 필수 너는 기록이 증명이다 gogs");
}

🎓 선생님의 세부 가이드 (14.0): "Lexer의 정교함"
 * Peeking: 때로는 다음 토큰을 확정하기 위해 한 글자를 미리 봐야 할 때가 있습니다. (예: =와 == 구분). 이를 위해 peek_char 메서드를 설계에 추가하는 것이 좋습니다.
 * 성능: Lexer는 모든 소스 글자를 최소 한 번은 훑어야 하므로 성능에 민감합니다. char 벡터 대신 바이트 스트림을 직접 다루는 방식도 고려해볼 수 있습니다.
 * 위치 추적: 나중에 에러 메시지를 띄울 때 "3행 5열에서 에러 발생"이라고 말해주려면, Lexer가 현재 몇 번째 행과 열을 읽고 있는지 기록해야 합니다.
