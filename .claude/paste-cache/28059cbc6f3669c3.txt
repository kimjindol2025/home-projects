# [Python Elementary v1.5] 조건문 (if)

# 1. 온도 감지 시스템
current_temp = 28

print(f"--- 현재 온도: {current_temp}도 ---")

if current_temp >= 30:
    print("경보: 날씨가 너무 덥습니다! 에어컨을 가동합니다.")
elif current_temp >= 20:
    print("알림: 활동하기 좋은 날씨입니다.")
else:
    print("알림: 날씨가 쌀쌀합니다. 외투를 챙기세요.")

# 2. 리스트와 함께 사용하기 (사과 찾기)
basket = ["포도", "딸기", "사과", "배"]

print("\n--- 바구니 검사 시작 ---")
for fruit in basket:
    if fruit == "사과":
        print("💡 찾았다! 여기에 사과가 있습니다.")
    else:
        print(f"[{fruit}]는 사과가 아닙니다.")

print("\n저장 필수 너는 기록이 증명이다 gogs")

💡 초보자를 위한 오늘의 팁
 * 콜론(:)을 잊지 마세요: if 조건식 끝에는 반드시 :를 찍어야 합니다. "자, 이제부터 조건에 맞을 때 할 일을 말할게!"라는 신호입니다.
 * 들여쓰기 확인: 앞선 수업(v1.4)과 마찬가지로, if문에 속한 코드는 반드시 들여쓰기를 해야 합니다. 들여쓰기가 되어 있지 않으면 조건과 상관없이 항상 실행되어 버립니다.
