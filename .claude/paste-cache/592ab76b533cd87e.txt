bash 스크립트: 저장소 자동 생성
#!/bin/bash

GOGS_TOKEN="your_token_here"
REPO_NAMES=("repo1" "repo2" "repo3")

for repo in "${REPO_NAMES[@]}"; do
  curl -X POST https://gogs.dclub.kr/api/v1/user/repos \
    -H "Authorization: token $GOGS_TOKEN" \
    -H "Content-Type: application/json" \
    -d "{
      \"name\": \"$repo\",
      \"description\": \"Auto-created repository\",
      \"private\": false
    }"
  echo "Created: $repo"
done
Node.js: 저장소 생성
const axios = require('axios');

const GOGS_TOKEN = 'your_token_here';
const GOGS_API = 'https://gogs.dclub.kr/api/v1';

async function createRepo(repoName, description = '') {
  try {
    const response = await axios.post(
      `${GOGS_API}/user/repos`,
      {
        name: repoName,
        description: description,
        private: false
      },
      {
        headers: {
          'Authorization': `token ${GOGS_TOKEN}`,
          'Content-Type': 'application/json'
        }
      }
    );
    console.log('✅ 저장소 생성:', response.data.html_url);
    return response.data;
  } catch (error) {
    console.error('❌ 오류:', error.response?.data?.message || error.message);
  }
}

createRepo('my-awesome-repo', 'My repository description');
Python: 저장소 검색
import requests
import json

GOGS_TOKEN = 'your_token_here'
GOGS_API = 'https://gogs.dclub.kr/api/v1'

def search_repos(keyword, limit=10):
    response = requests.get(
        f'{GOGS_API}/repos/search',
        params={'q': keyword, 'limit': limit},
        headers={'Authorization': f'token {GOGS_TOKEN}'}
    )

    if response.status_code == 200:
        data = response.json()
        for repo in data.get('data', []):
            print(f"📦 {repo['full_name']} - {repo['description']}")
    else:
        print(f"❌ 오류: {response.json()}")

search_repos('freelang')
