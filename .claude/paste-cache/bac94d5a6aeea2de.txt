🛠️ gogs 저장소 v3.9 설계: 최종 통합 리허설
클로드코드(학생)에게 제2장의 모든 정수를 쏟아부은 **'중간고사 대비 통합 엔진'**을 구현하게 하세요.
// [제2장: 흐름의 통제 v3.9 최종 완결]

#[derive(Debug, PartialEq)]
enum SystemStatus {
    Ready,
    Active,
    ShuttingDown,
    EmergencyStop(String),
}

fn main() {
    println!("--- v3.9 Final Logic Validation: The Grand Finale ---");

    // 1. 테스트 데이터: 정상, 경고, 에러, 유효하지 않은 값의 혼합
    let raw_inputs = vec![10, 85, 0, 120, 45, 99];
    let mut current_status = SystemStatus::Ready;
    let mut log_history = Vec::new();

    println!("v3.9 통합 무결성 검사 시작...");

    // 2. v3.8의 반복자와 v3.4의 패턴 매칭 결합
    'main_engine: for (cycle, &input) in raw_inputs.iter().enumerate() {
        println!("\n[Cycle {}] 입력값 분석: {}", cycle + 1, input);

        // v3.6 Guard: 이미 비상 정지 상태라면 즉시 종료
        if let SystemStatus::EmergencyStop(_) = current_status {
            break 'main_engine;
        }

        // v3.7 & v3.9: 상태 전이와 논리적 판별
        current_status = match (current_status, input) {
            // v3.1 & v3.4: 복합 조건 매칭 (Pattern Guard)
            (SystemStatus::Ready, _) if input > 0 => {
                println!("  -> 초기 기동 성공.");
                SystemStatus::Active
            }
            (SystemStatus::Active, i) if i >= 100 => {
                let msg = format!("임계치 초과(Value: {})", i);
                SystemStatus::EmergencyStop(msg)
            }
            (SystemStatus::Active, 0) => {
                println!("  [경고] 유효하지 않은 입력. 건너뜁니다.");
                SystemStatus::Active // 상태 유지
            }
            (SystemStatus::Active, i) => {
                let action = if i >= 80 { "고출력 모드" } else { "정상 모드" };
                println!("  -> 연산 수행 중: {}", action);
                SystemStatus::Active
            }
            (status, _) => status, // 나머지 상태 유지
        };

        log_history.push(format!("Cycle {}: {:?}", cycle + 1, current_status));

        // v3.3: 특정 조건 발생 시 루프 강제 제어
        if let SystemStatus::EmergencyStop(ref reason) = current_status {
            println!("  !! 비상 정지 발령: {}", reason);
            // 더 이상의 처리는 무의미하므로 즉시 엔진 중단
            break 'main_engine;
        }
    }

    // 3. 최종 결과 보고
    println!("\n==========================================");
    println!("v3.9 최종 시스템 리포트");
    for entry in log_history { println!("  - {}", entry); }
    println!("최종 시스템 상태: {:?}", current_status);
    println!("==========================================");

    println!("기록: 제2장 모든 논리적 설계가 상충 없이 완벽히 통합됨.");
    println!("저장 필수 너는 기록이 증명이다 gogs");
}
