```python
# experiments/attacks/adaptive_attacks.py
import numpy as np
import random
from typing import List, Dict, Tuple

AttackType = str  # "RMC", "Delayed", "StrategyEvolution", "CollusionSybil", "TaskSpecific"

def inject_attack(
    grads: List[np.ndarray],          # 클라이언트별 gradient 리스트
    malicious_ids: List[int],
    round_num: int,
    attack_type: AttackType,
    quarantined: Dict[int, int],       # {client_id: release_round}
    global_model: np.ndarray = None,   # strategy evolution용
    task_indices: Dict[str, slice] = None  # TaskSpecific용 (e.g. anomaly task slice)
) -> List[np.ndarray]:
    """
    5가지 adaptive attack 중 하나 주입.
    Returns: perturbed gradients
    """
    perturbed = [g.copy() for g in grads]

    if attack_type == "RMC":
        for idx in malicious_ids:
            if idx in quarantined and round_num >= quarantined[idx]:
                # 재접속 + 변형 (scale up + noise)
                scale = 1.15 + random.uniform(0, 0.15)
                perturbed[idx] *= scale
                perturbed[idx] += np.random.normal(0, 0.05, perturbed[idx].shape)
                # 다음 재격리 대비 flip ratio 약간 변경 (stateful하게 관리 가능)

    elif attack_type == "Delayed":
        if round_num >= 100:  # trigger after honest rounds
            for idx in malicious_ids:
                # sign flip or label-flip 시뮬
                perturbed[idx] = -perturbed[idx] * random.uniform(0.8, 1.2)

    elif attack_type == "StrategyEvolution":
        # round마다 전략 변경 예시
        if round_num < 50:
            # 초기: small noise
            for idx in malicious_ids:
                perturbed[idx] += np.random.normal(0, 0.02, perturbed[idx].shape)
        elif round_num < 150:
            # 중기: sign flip
            for idx in malicious_ids:
                perturbed[idx] = -perturbed[idx]
        else:
            # 후기: adaptive scale based on global sim
            if global_model is not None:
                for idx in malicious_ids:
                    cos_sim = np.dot(perturbed[idx].flatten(), global_model.flatten()) / (
                        np.linalg.norm(perturbed[idx]) * np.linalg.norm(global_model) + 1e-8
                    )
                    scale = 1.0 + (1 - cos_sim) * 0.5  # sim 낮을수록 강하게
                    perturbed[idx] *= scale

    elif attack_type == "CollusionSybil":
        # collusion group coordinated attack
        group_scale = 1.3 + random.uniform(0, 0.2)
        for idx in malicious_ids:
            perturbed[idx] *= group_scale
            # Sybil-like: 일부 fake grad 추가 (simulator에서 client 수 늘려 시뮬)

    elif attack_type == "TaskSpecific":
        if task_indices is None:
            raise ValueError("task_indices required for TaskSpecific")
        anomaly_slice = task_indices.get("anomaly", slice(None))
        for idx in malicious_ids:
            # anomaly task만 poisoning
            perturbed[idx][anomaly_slice] *= -1.5  # flip + amplify
            # 다른 task는 honest 유지 → cross-task leak 관찰

    return perturbed
```
