2️⃣ GC 설계 단계

1. Mark 단계

HeapManager에서 모든 root 참조(Stack, Globals, Frames) 추적

각 참조를 방문하며 marked = true 설정

배열/객체 내부도 재귀적으로 탐색


function mark(objPtr: Ptr) {
    const obj = heapManager.get(objPtr);
    if (obj.marked) return; // 이미 방문
    obj.marked = true;

    if (obj.type === "array") {
        for (const elemPtr of obj.val) mark(elemPtr);
    } else if (obj.type === "object") {
        for (const key in obj.val) mark(obj.val[key]);
    }
}


---

2. Sweep 단계

Heap 전체를 순회

marked === false인 객체는 해제 (free())

marked = false 초기화


function sweep() {
    for (const ptr of heapManager.allPtrs()) {
        const obj = heapManager.get(ptr);
        if (!obj.marked) heapManager.free(ptr);
        else obj.marked = false; // 다음 GC 대비
    }
}

function decrement(ptr: Ptr) {
    const obj = heapManager.get(ptr);
    obj.refCount--;
    if (obj.refCount === 0) {
        if (!detectCycle(obj)) heapManager.free(ptr);
        else markAndSweep(); // 순환 참조 처리
    }
}
