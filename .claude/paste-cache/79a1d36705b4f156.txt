#!/bin/bash

API_KEY="dclub-api-key-2025-secure"
API_URL="http://192.168.45.73:50400"

echo "=== 저장소 검색 ==="
curl -s -H "x-api-key: $API_KEY" \
  "$API_URL/api/gogs/search?q=top" | jq '.repos[] | .name'

# 2. 코드 검색
echo "=== 코드 검색 (topbilliards에서 express 검색) ==="
curl -s -H "x-api-key: $API_KEY" \
  "$API_URL/api/gogs/topbilliards/search?q=express&limit=5" | \
  jq -r '.results[] | "\(.file):\(.line) - \(.content)"'

# 3. 커밋 목록

const axios = require('axios');

const client = axios.create({
  baseURL: 'http://192.168.45.73:50400',
  headers: { 'x-api-key': 'dclub-api-key-2025-secure' }
});

  const { data } = await client.get('/api/gogs/search', {
    params: { q: query }
  });

};

  });

import requests

class GogsAPI:
    def __init__(self, base_url, api_key):
        self.base_url = base_url
        self.headers = {'x-api-key': api_key}

    def search_repos(self, query):
        response = requests.get(
            f'{self.base_url}/api/gogs/search',
            headers=self.headers,
            params={'q': query}
        )
        return response.json()['repos']

    def search_code(self, repo, query, limit=20):
        response = requests.get(
            f'{self.base_url}/api/gogs/{repo}/search',
            headers=self.headers,
            params={'q': query, 'limit': limit}
        )
        return response.json()['results']

# 사용 예
api = GogsAPI('http://192.168.45.73:50400', 'dclub-api-key-2025-secure')

repos = api.search_repos('top')
print(f"찾은 저장소: {len(repos)}개")

code = api.search_code('topbilliards', 'express')
print(f"코드 검색 결과: {len(code)}개")
