// src/type-system.ts
import { Type, Primitive } from "./types";
import { ASTNode } from "./ast";

/** -----------------------------
 * TypeChecker
 * ----------------------------- */
export class TypeChecker {
  private env: Map<string, Type> = new Map();

  constructor(env?: Map<string, Type>) {
    if (env) this.env = env;
  }

  check(node: ASTNode): Type {
    switch (node.kind) {
      case "Literal": return this.checkLiteral(node);
      case "Variable": return this.checkVariable(node);
      case "BinaryOp": return this.checkBinaryOp(node);
      case "Call": return this.checkCall(node);
      default: return { kind: "any" };
    }
  }

  private checkLiteral(node: ASTNode): Type {
    switch (typeof node.value) {
      case "number": return { kind: "primitive", name: "num" };
      case "string": return { kind: "primitive", name: "str" };
      case "boolean": return { kind: "primitive", name: "bool" };
      default: return { kind: "any" };
    }
  }

  private checkVariable(node: ASTNode): Type {
    const t = this.env.get(node.name);
    if (!t) throw new Error(`Undefined variable: ${node.name}`);
    return t;
  }

  private checkBinaryOp(node: ASTNode): Type {
    const left = this.check(node.left);
    const right = this.check(node.right);
    if (left.kind === "primitive" && left.name === "num" &&
        right.kind === "primitive" && right.name === "num") {
      return { kind: "primitive", name: "num" };
    }
    throw new Error(`Type mismatch in binary op: ${JSON.stringify(left)} vs ${JSON.stringify(right)}`);
  }

  private checkCall(node: ASTNode): Type {
    const fnType = this.check(node.callee);
    if (fnType.kind !== "function") throw new Error("Call of non-function");
    if (fnType.args.length !== node.args.length) throw new Error("Argument count mismatch");
    for (let i = 0; i < fnType.args.length; i++) {
      const argType = this.check(node.args[i]);
      if (!this.matchType(argType, fnType.args[i])) throw new Error(`Argument ${i} type mismatch`);
    }
    return fnType.ret;
  }

  private matchType(t1: Type, t2: Type): boolean {
    if (t2.kind === "any") return true;
    if (t1.kind !== t2.kind) return false;
    switch (t1.kind) {
      case "primitive": return t1.name === (t2 as any).name;
      case "array": return this.matchType(t1.elem, (t2 as any).elem);
      case "object":
        const keys1 = Object.keys(t1.props);
        const keys2 = Object.keys((t2 as any).props);
        if (keys1.length !== keys2.length) return false;
        return keys1.every(k => this.matchType(t1.props[k], (t2 as any).props[k]));
      case "function":
        const f2 = t2 as any;
        if (t1.args.length !== f2.args.length) return false;
        for (let i = 0; i < t1.args.length; i++)
          if (!this.matchType(t1.args[i], f2.args[i])) return false;
        return this.matchType(t1.ret, f2.ret);
      default: return false;
    }
  }
}

/** -----------------------------
 * TypeInference
 * ----------------------------- */
export class TypeInference {
  constructor(private checker: TypeChecker) {}

  infer(node: ASTNode): Type {
    try {
      return this.checker.check(node);
    } catch {
      return { kind: "any" };
    }
  }
}

/** -----------------------------
 * TypeConstraints
 * ----------------------------- */
export class TypeConstraints {
  private constraints: Array<[Type, Type]> = [];

  add(t1: Type, t2: Type) {
    this.constraints.push([t1, t2]);
  }

  validate() {
    for (const [t1, t2] of this.constraints) {
      if (!this.matchType(t1, t2))
        throw new Error(`Type constraint violation: ${JSON.stringify(t1)} vs ${JSON.stringify(t2)}`);
    }
  }

  private matchType(t1: Type, t2: Type): boolean {
    if (t1.kind === "any" || t2.kind === "any") return true;
    if (t1.kind !== t2.kind) return false;
    switch (t1.kind) {
      case "primitive": return t1.name === (t2 as any).name;
      case "array": return this.matchType(t1.elem, (t2 as any).elem);
      case "object":
        const keys1 = Object.keys(t1.props);
        const keys2 = Object.keys((t2 as any).props);
        if (keys1.length !== keys2.length) return false;
        return keys1.every(k => this.matchType(t1.props[k], (t2 as any).props[k]));
      default: return false;
    }
  }
}
