## 🚀 분석 자동화 스크립트

### Bash 스크립트 예제

```bash
#!/bin/bash
# c11_repos_analysis.sh

REPOS=(
  "range-c" "moment-c" "knex-c" "lodash-c" "mime-c"
  "uuid-c" "fs-c" "glob-c" "chalk-c" "async-c"
  "graphql-c" "final-c" "connect-c" "cookie-c"
  "cookie_crypto-c" "cookie-parser-c" "body-c"
  "commander-c" "express-c" "c-libs-ci"
)

WORK_DIR="./c11_repos_analysis"
mkdir -p "$WORK_DIR"

for repo in "${REPOS[@]}"; do
  echo "Analyzing: $repo"

  # Clone repository
  git clone https://gogs.dclub.kr/kim/$repo.git "$WORK_DIR/$repo"

  cd "$WORK_DIR/$repo"

  # Count files
  C_FILES=$(find . -name "*.c" | wc -l)
  H_FILES=$(find . -name "*.h" | wc -l)
  C_LINES=$(find . -name "*.c" -exec wc -l {} + | tail -1 | awk '{print $1}')

  # Check files
  HAS_README="$([ -f README.md ] && echo 'YES' || echo 'NO')"
  HAS_TESTS="$([ -d tests ] && echo 'YES' || echo 'NO')"
  HAS_MAKEFILE="$([ -f Makefile ] && echo 'YES' || echo 'NO')"

  # Last commit
  LAST_COMMIT=$(git log -1 --format="%h %s" 2>/dev/null || echo "N/A")

  # Print results
  echo "$repo,$C_FILES,$H_FILES,$C_LINES,$HAS_README,$HAS_TESTS,$HAS_MAKEFILE,$LAST_COMMIT"

  cd -
done
```
